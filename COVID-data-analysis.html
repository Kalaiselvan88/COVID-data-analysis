df_age = df[~(df['age_group'] == 'Unknown')]

def calculateMean(ageGroup):
    if(ageGroup == '80'):
        return int((80+100)/2)
    else:
        age_list = ageGroup.split('-')
        age_list = [int(i) for i in age_list]
        return sum(age_list)/len(age_list)
